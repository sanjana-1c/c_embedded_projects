#include <stdio.h>
#include <unistd.h>  // for sleep function (seconds)

int main() {
    int times;
    
    printf("Enter number of times to blink LED: ");
    scanf("%d", &times);

    for(int i = 0; i < times; i++) {
        printf("LED ON\n");
        sleep(1);
        printf("LED OFF\n");
        sleep(1);
    }

    printf("Blinking finished!\n");
    return 0;
}
